// Action
dispatchShardHunt: (packId: string, prodAff: number, goalType: string) => {
  set((state) => ({ packRegistry: state.packRegistry.map(p => p.id === packId ? { ...p, status: 'hunting' } : p) }));
  ecosystem.dispatchShardHunt(packId, prodAff, goalType);
},

// Getter
